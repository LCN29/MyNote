# MySQL 存储过程

```sql
DELIMITER $$

CREATE PROCEDURE demo_in_parameter()

BEGIN
	declare str varchar(50);
	declare i int;
	
	set i = 1;
	
	while i < 6 do
	
		set str = CONCAT("测试", i);
	
		set i = i + 1;
		
		select str;
	
	end while;

	select i;
	
END;  
$$
DELIMITER ;

# 执行存储过程
call demo_in_parameter();

# 删除存储过程
drop PROCEDURE demo_in_parameter;
```
